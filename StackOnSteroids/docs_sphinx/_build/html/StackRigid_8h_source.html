109 static uint32_t adlerChecksum(const void* firstBlock, size_t len) {
110  uint32_t a = 1, b = 0;
111  const uint32_t MOD_ADLER = 65521;
112  for (size_t index = 0; index < len; ++index) {
113  a = (a + ((char*)firstBlock)[index]) % MOD_ADLER;
114  b = (b + a) % MOD_ADLER;
115  }
116  return (b << 16) | a;
